 5 | typedef FutureOrWidgetBuilder<T> = Widget Function(
 6 |     BuildContext context, AsyncSnapshot<T> snapshot);
 7 | 
 8 | class FutureOrBuilder<T> extends StatelessWidget {
 9 |   final FutureOr<T> future;
10 |   final FutureOrWidgetBuilder<T> builder;
11 |   final T? initialValue;
12 | 
13 |   const FutureOrBuilder({
14 |     super.key,
15 |     required this.future,
16 |     required this.builder,
17 |     this.initialValue,
18 |   });
19 | 
20 |   @override
21 |   Widget build(BuildContext context) {
22 |     if (future is Future<T>) {
23 |       return FutureBuilder<T>(
24 |         future: future as Future<T>,
25 |         initialData: initialValue,
26 |         builder: builder,
27 |       );
28 |     }
29 |     return builder(
30 |         context, AsyncSnapshot.withData(ConnectionState.done, future as T));
31 |   }
32 | }
